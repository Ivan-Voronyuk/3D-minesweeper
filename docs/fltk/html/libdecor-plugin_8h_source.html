61struct libdecor_plugin_description {
62 /* API version the plugin is compatible with. */
63 int api_version;
64
65 /* Human readable string describing the plugin. */
66 char *description;
67
68 /* A plugin has a bitmask of capabilities. The plugin loader can use this
69 * to load a plugin with the right capabilities. */
70 enum libdecor_plugin_capabilities capabilities;
71
72 /*
73 * The priorities field points to a list of per desktop priorities.
74 * properties[i].desktop is matched against XDG_CURRENT_DESKTOP when
75 * determining what plugin to use. The last entry in the list MUST have
76 * the priorities[i].desktop pointer set to NULL as a default
77 * priority.
78 */
79 const struct libdecor_plugin_priority *priorities;
80
81 /* Vfunc used for constructing a plugin instance. */
82 libdecor_plugin_constructor constructor;
83
84 /* NULL terminated list of incompatible symbols. */
85 char *conflicting_symbols[1024];
86};

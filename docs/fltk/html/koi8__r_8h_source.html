115static int
116koi8_r_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
117{
118 (void)conv; (void)n;
119 unsigned char c = 0;
120 if (wc < 0x0080) {
121 *r = wc;
122 return 1;
123 }
124 else if (wc >= 0x00a0 && wc < 0x00f8)
125 c = koi8_r_page00[wc-0x00a0];
126 else if (wc >= 0x0400 && wc < 0x0458)
127 c = koi8_r_page04[wc-0x0400];
128 else if (wc >= 0x2218 && wc < 0x2268)
129 c = koi8_r_page22[wc-0x2218];
130 else if (wc >= 0x2320 && wc < 0x2328)
131 c = koi8_r_page23[wc-0x2320];
132 else if (wc >= 0x2500 && wc < 0x25a8)
133 c = koi8_r_page25[wc-0x2500];
134 if (c != 0) {
135 *r = c;
136 return 1;
137 }
138 return RET_ILSEQ;
139}

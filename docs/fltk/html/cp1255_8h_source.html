34static int
35cp1255_mbtowc (conv_t conv, ucs4_t *pwc, const unsigned char *s, int n)
36{
37 unsigned char c = *s;
38 if (c < 0x80) {
39 *pwc = (ucs4_t) c;
40 return 1;
41 }
42 else {
43 unsigned short wc = cp1255_2uni[c-0x80];
44 if (wc != 0xfffd) {

48 }

92static int
93cp1255_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
94{
95 unsigned char c = 0;
96 if (wc < 0x0080) {
97 *r = wc;
98 return 1;
99 }
100 else if (wc >= 0x00a0 && wc < 0x00f8)
101 c = cp1255_page00[wc-0x00a0];
102 else if (wc == 0x0192)
103 c = 0x83;
104 else if (wc >= 0x02c0 && wc < 0x02e0)
105 c = cp1255_page02[wc-0x02c0];
106 else if (wc >= 0x05b0 && wc < 0x05f8)
107 c = cp1255_page05[wc-0x05b0];
108 else if (wc >= 0x2008 && wc < 0x2040)
109 c = cp1255_page20[wc-0x2008];
110 else if (wc == 0x20aa)
111 c = 0xa4;
112 else if (wc == 0x20ac)
113 c = 0x80;
114 else if (wc == 0x2122)
115 c = 0x99;
116 if (c != 0) {
117 *r = c;
118 return 1;
119 }
120 return RET_ILSEQ;
121}

77static int
78iso8859_13_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
79{
80 (void)conv; (void)n;
81 unsigned char c = 0;
82 if (wc < 0x00a0) {
83 *r = wc;
84 return 1;
85 }
86 else if (wc >= 0x00a0 && wc < 0x0180)
87 c = iso8859_13_page00[wc-0x00a0];
88 else if (wc >= 0x2018 && wc < 0x2020)
89 c = iso8859_13_page20[wc-0x2018];
90 if (c != 0) {
91 *r = c;
92 return 1;
93 }
94 return RET_ILSEQ;
95}

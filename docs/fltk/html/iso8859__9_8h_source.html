54iso8859_9_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
55{
56 (void)conv; (void)n;

63 c = iso8859_9_page00[wc-0x00d0];
64 else if (wc >= 0x0118 && wc < 0x0160)

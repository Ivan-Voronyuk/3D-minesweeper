158 // Scaling and clipping stuff for internal usage.
159 // Currently specific to Fl_Xlib_Graphics_Driver (16-bit coordinate clipping)
160 // These methods must not be virtual so they can be inlined.
161 // All these methods work with "real" drawing coordinates (after scaling).
162
163 // Clipping limits
164 int clip_max() { return clip_max_; }
165 int clip_min() { return -clip_max_; }
166
167 /*
168 clip_xy() returns a single coordinate value clipped to the 16-bit
169 coordinate space.
170
171 This can be used to draw horizontal and vertical lines that can be
172 handled by X. Each single coordinate value can be clipped individually
173 and the result can be used directly, e.g. in fl_xyline() and fl_yxline().
174 */
175 int clip_xy(int x) {
176 return (x < clip_min() ? clip_min() : (x > clip_max() ? clip_max() : x ));
177 }

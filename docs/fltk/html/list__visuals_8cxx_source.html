80static void print_mask(XVisualInfo* p) {
81 int n = 0;
82 int what = 0;
83 int print_anything = 0;
84 char buf[20];
85 char *q = buf;
86 *q = 0;
87 int b; unsigned int m; for (b=32,m=0x80000000; ; b--,m>>=1) {
88 int new_what = 0;
89 if (p->red_mask&m) new_what = 'r';
90 else if (p->green_mask&m) new_what = 'g';
91 else if (p->blue_mask&m) new_what = 'b';
92 else new_what = '?';
93 if (new_what != what) {
94 if (what && (what != '?' || print_anything)) {
95 q += snprintf(q, sizeof(buf) - (q-buf), "%d%c", n, what);
96 print_anything = 1;
97 }
98 what = new_what;
99 n = 1;
100 } else {
101 n++;
102 }
103 if (!b) break;
104 }
105 printf("%7s", buf);
106}

108void list_visuals() {
109 fl_open_display();
110 XVisualInfo vTemplate;
111 int num;
112 XVisualInfo *visualList = XGetVisualInfo(fl_display,0,&vTemplate,&num);
113
114 XPixmapFormatValues *pfvlist;
115 static int numpfv;
116 pfvlist = XListPixmapFormats(fl_display, &numpfv);
117
118 OverlayInfo *overlayInfo = 0;
119 int numoverlayinfo = 0;
120 Atom overlayVisualsAtom = XInternAtom(fl_display,"SERVER_OVERLAY_VISUALS",1);
121 if (overlayVisualsAtom) {
122 unsigned long sizeData, bytesLeft;
123 Atom actualType;
124 int actualFormat;
125 if (!XGetWindowProperty(fl_display, RootWindow(fl_display, fl_screen),
126 overlayVisualsAtom, 0L, 10000L, False,
127 overlayVisualsAtom, &actualType, &actualFormat,
128 &sizeData, &bytesLeft,
129 (unsigned char **) &overlayInfo))
130 numoverlayinfo = int(sizeData/4);
131 }
132
133#if HAVE_MULTIBUF
134 int event_base, error_base;
135 XmbufBufferInfo *mbuf, *sbuf;
136 int nmbuf = 0, nsbuf = 0;
137 if (XmbufQueryExtension(fl_display,&event_base, &error_base)) {
138 XmbufGetScreenInfo(fl_display,RootWindow(fl_display,fl_screen),
139 &nmbuf, &mbuf, &nsbuf, &sbuf);
140 }
141#endif
142
143 for (int i=0; i<num; i++) {
144 XVisualInfo *p = visualList+i;
145
146 XPixmapFormatValues *pfv;
147 for (pfv = pfvlist; ; pfv++) {
148 if (pfv >= pfvlist+numpfv) {pfv = 0; break;} // should not happen!
149 if (pfv->depth == p->depth) break;
150 }
151
152 int j = pfv ? pfv->bits_per_pixel : 0;
153 printf(" %2ld: %s %2d/%d", p->visualid, ClassNames[p->c_class],
154 p->depth, j);
155 if (j < 10) putchar(' ');
156
157 print_mask(p);
158
159 for (j=0; j<numoverlayinfo; j++) {
160 OverlayInfo *o = &overlayInfo[j];
161 if (o->overlay_visual == long(p->visualid)) {
162 printf(" overlay(");
163 if (o->transparent_type==1) printf("transparent pixel %ld, ",o->value);
164 else if (o->transparent_type==2) printf("transparent mask %ld, ",o->value);
165 else printf("opaque, ");
166 printf("layer %ld)", o->layer);
167 }
168 }
169
170#if HAVE_MULTIBUF
171 for (j=0; j<nmbuf; j++) {
172 XmbufBufferInfo *m = &mbuf[j];
173 if (m->visualid == p->visualid)
174 printf(" multibuffer(%d)", m->max_buffers);
175 }
176 for (j=0; j<nsbuf; j++) {
177 XmbufBufferInfo *m = &sbuf[j];
178 if (m->visualid == p->visualid)
179 printf(" stereo multibuffer(%d)", m->max_buffers);
180 }
181#endif
182
183 if (p->visualid==XVisualIDFromVisual(DefaultVisual(fl_display,fl_screen)))
184 printf(" (default visual)");
185
186 putchar('\n');
187 }
188 if ( overlayInfo ) { XFree(overlayInfo); overlayInfo = 0; }
189}

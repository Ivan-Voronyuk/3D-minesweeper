108static int
109cp1256_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
110{
111 unsigned char c = 0;
112 if (wc < 0x0080) {
113 *r = wc;
114 return 1;
115 }
116 else if (wc >= 0x00a0 && wc < 0x0100)
117 c = cp1256_page00[wc-0x00a0];
118 else if (wc >= 0x0150 && wc < 0x0198)
119 c = cp1256_page01[wc-0x0150];
120 else if (wc == 0x02c6)
121 c = 0x88;
122 else if (wc >= 0x0608 && wc < 0x06d8)
123 c = cp1256_page06[wc-0x0608];
124 else if (wc >= 0x2008 && wc < 0x2040)
125 c = cp1256_page20[wc-0x2008];
126 else if (wc == 0x20ac)
127 c = 0x80;
128 else if (wc == 0x2122)
129 c = 0x99;
130 if (c != 0) {
131 *r = c;
132 return 1;
133 }
134 return RET_ILSEQ;

157class Fl_Input_Choice_Type : public Fl_Menu_Manager_Type
158{
159 typedef Fl_Menu_Manager_Type super;
160 int textstuff(int w, Fl_Font& f, int& s, Fl_Color& c) FL_OVERRIDE {
161 Fl_Input_Choice *myo = (Fl_Input_Choice*)(w==4 ? ((Fl_Widget_Type*)this->factory)->o : this->o);
162 switch (w) {
163 case 4:
164 case 0: f = (Fl_Font)myo->textfont(); s = myo->textsize(); c = myo->textcolor(); break;
165 case 1: myo->textfont(f); break;

167 case 3: myo->textcolor(c); break;
168 }
169 return 1;
170 }
